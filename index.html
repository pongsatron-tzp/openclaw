🔒 SECURITY PRINCIPLES (หลักความปลอดภัย)
🔑
API Keys ใน .env
ไม่ hardcode ใน skill file — ใช้ env vars เท่านั้น ไม่ commit .env
👤
DB User จำกัดสิทธิ์
สร้าง user แยกสำหรับ OpenClaw — ให้แค่ INSERT ไม่ให้ DROP/DELETE
🚦
Skill Boundary
แต่ละ Skill ทำได้แค่งานเดียว ไม่ให้ Skill เข้าถึง credentials ข้าม scope
📵
No Bidirectional DB
OpenClaw อ่าน DB ไม่ได้ — write-only ป้องกัน data leakage ผ่าน AI
🕐
Cron + Rate Limit
รัน schedule แทน realtime polling ลด API call และป้องกัน loop
📝
Audit Log
ทุก INSERT มี timestamp + source_skill field ตรวจสอบย้อนหลังได้

// skills/fetch-summarize.skill.js // OpenClaw Skill — อ่าน API + สรุป (ไม่ write ข้อมูลใดๆ) export default { name: "fetch-and-summarize", description: "Fetch newsletters/APIs and summarize with AI", async run({ ai, env }) { // ✅ READ ONLY — ดึงข้อมูลจาก free API const res = await fetch( `https://hacker-news.firebaseio.com/v0/topstories.json` ); const ids = (await res.json()).slice(0, 5); // ดึงรายละเอียดแต่ละข่าว const stories = await Promise.all( ids.map(id => fetch( `https://hacker-news.firebaseio.com/v0/item/${id}.json` ).then(r => r.json())) ); // ✅ สรุปผ่าน AI (local — ไม่ส่งออกนอก) const summary = await ai.summarize( stories.map(s => `${s.title}: ${s.url}`).join("\n"), { format: "newsletter", lang: "th" } ); // ✅ return ออกไปให้ skill ถัดไปจัดการ (ไม่ write เอง) return { stories, summary, fetchedAt: new Date().toISOString() }; } };
// skills/save-to-db.skill.js // One-way write — INSERT only, ใช้ user ที่จำกัดสิทธิ์ import { Pool } from "pg"; // ✅ credentials จาก env ไม่ hardcode const db = new Pool({ connectionString: process.env.OPENCLAW_DB_URL, // DB user นี้มีแค่ INSERT privilege เท่านั้น }); export default { name: "save-newsletter", description: "Save summarized newsletter to DB (INSERT only)", async run({ input }) { const { summary, stories, fetchedAt } = input; // ✅ Validate ก่อน insert — ป้องกัน injection if (typeof summary !== "string" || summary.length > 10000) { throw new Error("Invalid summary data"); } // ✅ Parameterized query — ป้องกัน SQL injection await db.query( `INSERT INTO newsletters (summary, story_count, fetched_at, source_skill) VALUES ($1, $2, $3, $4)`, [summary, stories.length, fetchedAt, "fetch-and-summarize"] ); // ✅ ไม่มี SELECT, UPDATE, DELETE — write-only return { saved: true, at: fetchedAt }; } };
-- สร้าง table สำหรับรับข้อมูล CREATE TABLE newsletters ( id SERIAL PRIMARY KEY, summary TEXT NOT NULL, story_count INT, fetched_at TIMESTAMPTZ DEFAULT NOW(), source_skill VARCHAR(100), created_at TIMESTAMPTZ DEFAULT NOW() ); -- ✅ สร้าง user แยก — ให้ INSERT เท่านั้น CREATE USER openclaw_writer WITH PASSWORD 'strong-random-password'; -- ❌ ไม่ให้ SELECT (ป้องกัน data leak ผ่าน AI) REVOKE ALL ON newsletters FROM openclaw_writer; -- ✅ ให้แค่ INSERT GRANT INSERT ON newsletters TO openclaw_writer; GRANT USAGE ON SEQUENCE newsletters_id_seq TO openclaw_writer; -- ใน .env ของ OpenClaw -- OPENCLAW_DB_URL=postgres://openclaw_writer:password@localhost/mydb
